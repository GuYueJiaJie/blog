## 双指针

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} head
 * @return {boolean}
 */
var hasCycle = function(head) {
  if (head === null) return false;
  let slow = head,
    fast = head;
  while (fast !== null) {
    slow = slow.next;
    fast = fast.next;
    if (fast !== null) {
      fast = fast.next;
    } else {
      return false;
    }
    if (slow === fast) {
      return true;
    }
  }
  return false;
};
```
