## 动态规划

关键在于保存两天前的利润值。

```javascript
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
  // 关键在于保存两天前的那个dp_0
  let dp_0 = 0,
    dp_1 = -Infinity;
  let temp = 0;

  for (let i = 0; i < prices.length; i++) {
    let temp1 = dp_0;
    dp_0 = Math.max(dp_0, dp_1 + prices[i]);
    dp_1 = Math.max(dp_1, temp - prices[i]);
    temp = temp1;
  }

  return dp_0;
};
```
