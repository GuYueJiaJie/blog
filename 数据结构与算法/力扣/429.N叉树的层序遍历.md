## 层序遍历

```javascript
/**
 * // Definition for a Node.
 * function Node(val,children) {
 *    this.val = val;
 *    this.children = children;
 * };
 */
/**
 * @param {Node} root
 * @return {number[][]}
 */
var levelOrder = function(root) {
  if (root === null) return [];
  let res = [];
  let queue = [root];

  while (queue.length > 0) {
    let length = queue.length;
    let tmp = [];
    for (let i = 0; i < length; i++) {
      let cur = queue.shift();
      tmp.push(cur.val);
      queue.push(...cur.children);
    }
    res.push(tmp);
  }
  return res;
};
```
