## 栈

```javascript
/**
 * @param {string[]} ops
 * @return {number}
 */
var calPoints = function(ops) {
  let sum = 0;
  let stack = [];
  for (let i = 0; i < ops.length; i++) {
    let point = ops[i],
      cur;
    if (point === "+") {
      cur = stack[stack.length - 1] + stack[stack.length - 2];
      stack.push(cur);
      sum += cur;
    } else if (point === "D") {
      cur = stack[stack.length - 1] * 2;
      stack.push(cur);
      sum += cur;
    } else if (point === "C") {
      cur = stack.pop();
      sum -= cur;
    } else {
      cur = Number(point);
      stack.push(cur);
      sum += cur;
    }
  }
  return sum;
};
```
